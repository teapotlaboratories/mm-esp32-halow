135static inline char mm_nibble_to_hex_char(uint8_t nibble)
136{
137 nibble &= 0x0f;
138 if (nibble < 0x0a)
139 {
140 return '0' + nibble;
141 }
142 else
143 {
144 return 'A' + nibble - 0x0a;
145 }
146}
